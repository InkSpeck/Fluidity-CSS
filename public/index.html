Box-Shadow Mixins:

shadow(x,y,b,c) / inshadow(x,y,b,s,c) / box-shadow()

Stylus
// A single box shadow #example1 shadow(0,0,9px,#000000) // A single inset shadow #example2 inshadow(0,0,9px,0px,#000000) // Need more than one? No Problem use the full syntax. #example box-shadow inset 0 0 6px 0px #000, 0 1px 1px #ccc
rendered CSS
#example1 { -moz-box-shadow: 0 0 9px #000; -webkit-box-shadow: 0 0 9px #000; -o-box-shadow: 0 0 9px #000; box-shadow: 0 0 9px #000; } #example2 { -webkit-box-shadow: inset 0 0 9px 0px #000; -moz-box-shadow: inset 0 0 9px 0px #000; -o-box-shadow: inset 0 0 9px 0px #000; box-shadow: inset 0 0 9px 0px #000; } #example3 { -moz-box-shadow: inset 0 0 6px 0px #000, 0 1px 1px #ccc; -webkit-box-shadow: inset 0 0 6px 0px #000, 0 1px 1px #ccc; -o-box-shadow: inset 0 0 6px 0px #000, 0 1px 1px #ccc; box-shadow: inset 0 0 6px 0px #000, 0 1px 1px #ccc; }

Gradient Mixins:

gradient(beginning, middle, end) / hgradient(beginning, middle, end) / rgradient(x,y,r,e)

Stylus
// A vertical 3 color gradient from top to bottom // 3 colors insures smoother gradients with less banding #example1 gradient #fff #ccc #000 // A 3 color horizontal gradient #example2 hgradient #fff #ccc #000 // A Radial Gradient - options (pos x, pos y, radial color, end color) #example3 rgradient(80%, 0%, #fff ,#000)
rendered CSS
#example1 { background-color: #000; background-image: -webkit-gradient(linear, left bottom, left top, color-stop(0.19, #000), color-stop(0.64, #ccc), color-stop(1, #fff)); background-image: -moz-linear-gradient(center bottom, #000 19%, #ccc 64%, #fff 100%); background-image: -o-linear-gradient(top, #fff, #000); background-image: linear-gradient(center bottom, #000 19%, #ccc 64%, #fff 100%); } #example2 { background-color: #000; background-image: -webkit-gradient(linear, left top, right top, color-stop(0.19, #000), color-stop(0.64, #ccc), color-stop(1, #fff)); background-image: -moz-linear-gradient(left center, #000 19%, #ccc 64%, #fff 100%); background-image: -o-linear-gradient(left, #fff, #000); background-image: linear-gradient(left center, #000 19%, #ccc 64%, #fff 100%); } #example3 { background-color: #000; background-image: -moz-radial-gradient(80% 0%, circle farthest-side, #fff, #000 100%); background-image: -webkit-gradient(radial, 80% 0%, 0px, 80% 0%, 960px, from(#fff), to(#000)); background-image: -webkit-radial-gradient(80% 0%, circle, #fff, #000); background-image: -o-radial-gradient(80% 0%, circle farthest-side, #fff, #000 100%); background-image: radial-gradient(80% 0%, circle farthest-side, #fff, #000 100%); }
